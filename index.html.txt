    <!-- Script / تفاعل السلة والطلب السريع -->
    <script>
        let cart = [];
        const cartSidebar = document.getElementById('cart-sidebar');
        const cartCount = document.getElementById('cart-count');
        const cartItemsContainer = document.getElementById('cart-items');
        const cartTotal = document.getElementById('cart-total');
        const emptyCartMsg = document.getElementById('empty-cart-msg');

        function toggleCart() {
            cartSidebar.classList.toggle('-translate-x-full');
        }

        function addToCart(name, price) {
            const existingItem = cart.find(item => item.name === name);
            if (existingItem) {
                existingItem.qty += 1;
            } else {
                cart.push({ name, price, qty: 1 });
            }
            updateCart();
            
            if(cartSidebar.classList.contains('-translate-x-full')) {
                toggleCart();
            }
        }

        function removeFromCart(name) {
            cart = cart.filter(item => item.name !== name);
            updateCart();
        }

        function updateCart() {
            const totalItems = cart.reduce((acc, item) => acc + item.qty, 0);
            cartCount.textContent = totalItems;

            if (cart.length === 0) {
                emptyCartMsg.style.display = 'block';
                cartItemsContainer.querySelectorAll('.cart-item').forEach(el => el.remove());
                cartTotal.textContent = "0 ريال";
                return;
            }

            emptyCartMsg.style.display = 'none';
            cartItemsContainer.querySelectorAll('.cart-item').forEach(el => el.remove());

            let totalCost = 0;

            cart.forEach(item => {
                totalCost += item.price * item.qty;
                const itemEl = document.createElement('div');
                itemEl.className = 'cart-item flex justify-between items-center bg-white/5 p-4 rounded-xl border border-white/5 transition duration-300 hover:border-red-500/40';
                itemEl.innerHTML = `
                    <div>
                        <h4 class="font-bold text-white">${item.name}</h4>
                        <p class="text-sm text-gray-400 mt-1">${item.price} ريال × ${item.qty}</p>
                    </div>
                    <div class="flex items-center gap-4">
                        <span class="font-bold text-red-500 font-serif">${item.price * item.qty} ريال</span>
                        <button onclick="removeFromCart('${item.name}')" class="text-red-500 hover:text-red-400 p-1">
                            <i class="fa-solid fa-trash-can"></i>
                        </button>
                    </div>
                `;
                cartItemsContainer.appendChild(itemEl);
            });

            cartTotal.innerHTML = `${totalCost} <span class="text-sm text-white">ريال</span>`;
        }

        function sendOrder() {
            if (cart.length === 0) {
                alert("الحقيبة فارغة، أضف بعض الوجبات القاتلة أولاً!");
                return;
            }

            let message = "⚔️ *طلب تسليح جديد من مطعم تنتاكي x غارة الليل* ⚔️\n\n";
            cart.forEach((item, index) => {
                message += `${index + 1}. *${item.name}* (الكمية: ${item.qty}) -> ${item.price * item.qty} ريال يمني\n`;
            });
            
            const total = cart.reduce((acc, item) => acc + (item.price * item.qty), 0);
            message += `\n💰 *إجمالي الغنائم الكلي:* ${total} ريال يمني\n\n🎯 يرجى إرسال موقعك لبدء هجوم التوصيل الفوري!`;

            const encodedMessage = encodeURIComponent(message);
            // ضع رقم الواتساب الخاص بك بالصيغة الدولية لتلقي الطلبات مباشرة
            const whatsappNumber = "967770000000"; 
            window.open(`https://wa.me/${whatsappNumber}?text=${encodedMessage}`, '_blank');
        }
    </script>
